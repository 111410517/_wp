## 1. 習題一：個人介紹網頁 (aboutme.html)

**設計理念與技術實作：**
這是我這學期的第一個作業，目的是學習 HTML 標籤結構與 CSS 排版。我設計了一個去圓角、使用 2px 粗黑線條（2px solid #111827）的極簡幾何風版面。

## 3. 習題三：Node.js 環境初探 (hello.js)

**設計理念與技術實作：**
這是第一個在 Node.js 本地環境下執行的 JS 程式。主要學習目標是理解 Node.js 與瀏覽器 runtime 的本質差異：
- **環境差異**：Node.js 的執行環境沒有瀏覽器的 DOM/BOM 物件（沒有 `window`、`document`、`alert`），但具備系統核心 API，可以直接操作作業系統磁碟與檔案。
- **代碼實作**：定義了一個接受學生基本資料物件的 `printWelcomeMessage` 函數，使用 `console.log` 輸出精緻的文字邊框與學籍資料。

**完整程式碼 (`homework/03/hello.js`)：**
```javascript
/**
 * @file homework/03/hello.js
 * @description 國立金門大學資訊工程學系一年級首個 JavaScript 程式
 * @student 洪晨祐 (111410517)
 */

'use strict';

function printWelcomeMessage(info) {
    if (!info || typeof info !== 'object') {
        throw new TypeError("必須傳入有效的資訊物件");
    }
    
    const { school, department, grade, name, id } = info;
    if (!school || !department || !grade || !name || !id) {
        throw new TypeError("資訊物件欄位不完整");
    }

    const divider = "=".repeat(50);
    console.log(divider);
    console.log(`🚀 hello, world!`);
    console.log(`🏫 學校系所：${school} ${department} ${grade}`);
    console.log(`👤 學生姓名：${name} (${id})`);
    console.log(`📝 備註資訊：這是我的第一個 JavaScript 程式輸出！`);
    console.log(divider);
}

try {
    printWelcomeMessage({
        school: "國立金門大學",
        department: "資訊工程學系",
        grade: "一年級",
        name: "洪晨祐",
        id: "111410517"
```

## 5. 習題五：Express + EJS 全棧網誌系統 (blog_basic)

**設計理念與技術實作：**
這個作業實作了一個完整的 Express 多用戶部落格。其技術核心在於：
1. **sql.js 記憶體資料庫的持久化讀寫**：由於 sql.js 預設只將 SQLite 運作在記憶體中，一旦伺服器重啟所有資料都會消失。為了解決這個問題，我編寫了 `flushDatabaseToDisk` 函數。當使用者註冊或發布貼文時，程式會執行 `sqliteDatabaseInstance.export()` 將資料庫轉換為 Buffer，並透過 `fs.writeFileSync` 同步寫入硬碟的 `blog.db`，在啟動時再以 `fs.readFileSync` 載入，實現資料持久化。
2. **Session 登入狀態維持**：利用 `express-session` 中介軟體，並將使用者名稱 and 識別碼存在 `req.session` 中，以防止未登入用戶發布貼文。

**核心程式碼 (`homework/05/blog_basic/app.js`)：**

```javascript
const path = require('path');
const fs = require('fs');
const bcrypt = require('bcryptjs');
const session = require('express-session');

blogExpressApp.use(express.urlencoded({ extended: true }));
blogExpressApp.use(express.static(path.join(__dirname, 'public')));

blogExpressApp.use(session({
    secret: 'hong-chen-you-geek-secret-key-2026',
    resave: false,
    saveUninitialized: false,

}));

const DATABASE_FILE_PATH = path.join(__dirname, 'blog.db');

function runDatabaseCommand(sqlString, bindValues = []) {
    try {
        sqliteDatabaseInstance.run(sqlString, bindValues);
        flushDatabaseToDisk();
    } catch (dbError) {
        console.error(`[SQL 執行失敗]`, dbError.message);
        throw dbError;
    }
}

function queryDatabaseRows(sqlString, bindValues = []) {
    let statement = null;
    try {
        statement = sqliteDatabaseInstance.prepare(sqlString);
        statement.bind(bindValues);
        const fetchedRows = [];
        while (statement.step()) {
            fetchedRows.push(statement.getAsObject());
        }
        return fetchedRows;
    } catch (queryError) {
        console.error(`[SQL 查詢失敗]`, queryError.message);
        return [];
    } finally {
        if (statement) statement.free();
    }
}

function queryDatabaseSingle(sqlString, bindValues = []) {
    const queryResults = queryDatabaseRows(sqlString, bindValues);
    return queryResults.length > 0 ? queryResults[0] : null;
}

async function bootstrapDatabaseConnection() {
    try {
        const SQL = await initSqlJs();
        if (fs.existsSync(DATABASE_FILE_PATH)) {
            const databaseFileBuffer = fs.readFileSync(DATABASE_FILE_PATH);
            sqliteDatabaseInstance = new SQL.Database(databaseFileBuffer);
            console.log("✔ 已成功加載現有 SQLite 資料庫檔案。");
        } else {
            sqliteDatabaseInstance = new SQL.Database();
            console.log("✔ 已成功初始化全新的記憶體 SQLite 資料庫。");
        }

        runDatabaseCommand(`
            CREATE TABLE IF NOT EXISTS users (
                id INTEGER PRIMARY KEY AUTOINCREMENT,
                username TEXT UNIQUE NOT NULL,
                password TEXT NOT NULL,
                bio TEXT DEFAULT '',
                created_at DATETIME DEFAULT CURRENT_TIMESTAMP
            )
        `);

        runDatabaseCommand(`
            CREATE TABLE IF NOT EXISTS posts (
                id INTEGER PRIMARY KEY AUTOINCREMENT,
                user_id INTEGER NOT NULL,
                content TEXT NOT NULL,
                likes INTEGER DEFAULT 0,
                created_at DATETIME DEFAULT CURRENT_TIMESTAMP,
                FOREIGN KEY (user_id) REFERENCES users(id)
            )
        `);
    } catch (initError) {
        console.error("❌ 資料庫啟動異常：", initError.message);
    }
}

function flushDatabaseToDisk() {
    try {
        const exportedDatabaseData = sqliteDatabaseInstance.export();
        const diskFileBuffer = Buffer.from(exportedDatabaseData);
        fs.writeFileSync(DATABASE_FILE_PATH, diskFileBuffer);
    } catch (flushError) {
        console.error("❌ 資料庫刷入磁碟失敗：", flushError.message);
    }
}

blogExpressApp.use((req, res, next) => {
    res.locals.user = req.session.userId ? { id: req.session.userId, username: req.session.username } : null;
    next();
});

blogExpressApp.get('/', (req, res) => {
    const publishedArticles = queryDatabaseRows(`
        SELECT p.*, u.username as author_name 
        FROM posts p 
        JOIN users u ON p.user_id = u.id 
        ORDER BY p.created_at DESC
    `);
    res.render('index', { posts: publishedArticles });
});

blogExpressApp.post('/register', (req, res) => {
    const { username, password, confirmPassword } = req.body;
    if (!username || !password) return res.render('register', { error: '請填寫所有欄位' });
    if (password !== confirmPassword) return res.render('register', { error: '密碼不一致' });
    
    const existUser = queryDatabaseSingle('SELECT * FROM users WHERE username = ?', [username]);
    if (existUser) return res.render('register', { error: '此用戶名已被註冊' });
    
    const encryptedPasswordHash = bcrypt.hashSync(password, 10);
    runDatabaseCommand('INSERT INTO users (username, password) VALUES (?, ?)', [username, encryptedPasswordHash]);
    res.redirect('/login');
});

blogExpressApp.post('/login', (req, res) => {
    const { username, password } = req.body;
    const userInstance = queryDatabaseSingle('SELECT * FROM users WHERE username = ?', [username]);
    if (!userInstance || !bcrypt.compareSync(password, userInstance.password)) {
        return res.render('login', { error: '無效的帳號或密碼' });
    }
    req.session.userId = userInstance.id;
    req.session.username = userInstance.username;

});

blogExpressApp.post('/posts', (req, res) => {
    if (!req.session.userId) return res.redirect('/login');
    const { content } = req.body;
    if (content) {
        runDatabaseCommand('INSERT INTO posts (user_id, content) VALUES (?, ?)', [req.session.userId, content]);
```

```javascript
/** @type {number[]} */

 * 處理函數：演示傳址 (Reference) 的修改與重新賦值差異
 * @param {number[]} sequenceA - 傳入的陣列 A (將對其進行 In-place Mutation)

        throw new TypeError("傳入的參數 sequenceA 與 sequenceB 必須是陣列！");
    }

    // 1. 修改操作：直接改變 sequenceA 所指向的原始記憶體陣列內容 (就地修改)
    //    這會改變該 JSArray 對應的 FixedArray 內容，影響外部實參。
    sequenceA.push(99);

    // 2. 重新賦值操作：將 sequenceB 的本地參數指標指向一個全新的陣列 [100]
    //    這僅改變了當前 Activation Frame (棧幀) 內局部變數的指向，不干擾外部。
    sequenceB = [100];

console.log("dataSequenceBeta  ➔", dataSequenceBeta);  // [3, 4]

console.log("dataSequenceBeta  ➔", dataSequenceBeta);  // 保持 [3, 4]     ➔ 原始陣列未受影響

```

**Stack & Heap 記憶體模型指針分析：**
1. **初始狀態**：在 Stack 中宣告變數 `myArr`，它儲存的值是一個記憶體位址（指標），指向 Heap 中實際存放陣列 `[10, 20]` 的位置。
2. **屬性修改 (`mutateArray`)**：當 `myArr` 的位址作為引數傳給 `mutateArray(arr)` 時，Stack 中的參數 `arr` 也得到了同一個 Heap 陣列的位址。執行 `arr.push(999)` 會直接改動該 Heap 空間的內容，因此外部的 `myArr` 讀取時也會顯示修改後的內容。
3. **重新指派 (`reassignArray`)**：在 `reassignArray(arr)` 中，執行 `arr = [1, 2, 3]` 時，JS 會在 Heap 中建立一個全新的陣列 `[1, 2, 3]`，並把 Stack 中的局部參數 `arr` 的值（指標）改為這個新陣列的位址。此操作沒有變動外部 `myArr` 指向的原 Heap 空間，因此外部陣列依然保持原樣。

### 題 5. 實作「錯誤優先」的回呼函數 (Error-First Callback) (`challenge05_error_first.js`)
```javascript
// @ts-check

/**
 * 1. 異步安全回呼隊列 (AsyncSafeCallbackQueue)
 * 解決問題：防止 "Zalgo" (同步非同步混淆)，確保 Callback 永遠在下一個 Tick 執行，並隔離 Callback 內部崩潰。
 */
class AsyncSafeCallbackQueue {
    /**
     * 以異步且安全的方式執行 Error-First Callback
     * @param {Function} callback - 回呼函數
     * @param {Error|string|null} err - 錯誤物件或錯誤訊息
     * @param {*} data - 回傳的數據 Payload
     */
    static safeExecute(callback, err, data) {
        if (typeof callback !== 'function') {
            // 靜默防止非函數傳入導致的主執行緒崩潰
            return;
        }

        // 強制推遲到下一個 Event Loop Tick 執行 (釋放 Zalgo)
        setImmediate(() => {
            try {
                // 將字串錯誤標準化為 Error 物件，保留 Stack Trace
                const normalizedError = (err && typeof err === 'string') ? new Error(err) : err;
                callback(normalizedError, data);
            } catch (callbackInternalError) {
                // 【異常隔離】即使 Callback 內部代碼拋出未捕獲異常，也僅在此被捕獲隔離，絕不拖垮伺服器主行程
                console.error("🚨 [Callback Event Queue 異常隔離防護啟動] 偵測到 Callback 內部未捕獲異常:", callbackInternalError.stack);
            }
        });
    }
}

        return AsyncSafeCallbackQueue.safeExecute(resultCallback, "錯誤：userId 必須為正整數！", null);
    }
```

```javascript
    if (typeof resultCallback !== 'function') {
        throw new TypeError("必須提供有效的回呼函數以接收資料庫查詢結果！");
```

## 8. 期中專案：原生 Node.js 電商外設商城 (online_shop)

**設計理念與技術實作：**
這個專案的目標是不使用任何 Express 或其他框架，純用 Node.js 內建的 `http`、`fs` 與 `path` 模組來編寫一個靜態資源伺服器與購物車後端。我著重實作了以下兩個核心：
1. **MIME 類型對照字典**：瀏覽器在載入靜態檔案時，會嚴格要求伺服器回傳正確的 `Content-Type` 頭部資訊。我在 `server.js` 中建立了一個 `mimeTypes` 對照表，透過 `path.extname` 判斷副檔名。若是讀取圖片等二進位檔案，則不傳入 `utf-8` 編碼直接以 Binary 格式讀取，防止圖片資料損毀。
2. **目錄越界讀取 (Path Traversal) 防禦**：為了防止攻擊者利用 `../` 讀取伺服器上的敏感系統檔案（如 `etc/passwd`），我先用 `path.resolve` 算出絕對請求路徑與網站根目錄路徑。接著透過 `path.relative` 計算兩者的相對關係。若相對路徑以 `..` 開頭或為絕對路徑外的位置，便回傳 `403 Forbidden`。

**完整伺服器代碼 (`homework/08/online_shop/server.js`)：**
```javascript
// homework/08/online_shop/server.js
// 國立金門大學資工系一年級 洪晨祐 (111410517)
// 原生 Node.js 伺服器與目錄越界安全防範實作

const http = require('http');
const fs = require('fs');
const path = require('path');

const PORT = 3000;
const PUBLIC_DIR = path.resolve(__dirname);

const mimeTypes = {
    '.html': 'text/html; charset=utf-8',
    '.css': 'text/css; charset=utf-8',
    '.js': 'application/javascript; charset=utf-8',
    '.json': 'application/json; charset=utf-8',
    '.jpg': 'image/jpeg',
    '.png': 'image/png',
    '.ico': 'image/x-icon'
};

const server = http.createServer((req, res) => {
    if (req.method === 'POST' && req.url === '/api/checkout') {
        let body = '';
        req.on('data', chunk => body += chunk);
        req.on('end', () => {
            try {
                const cartItems = JSON.parse(body);
                let subtotal = cartItems.reduce((sum, item) => sum + (item.price * item.quantity), 0);
                let discount = subtotal >= 5000 ? 500 : 0;
                let total = subtotal - discount;

                res.writeHead(200, { 'Content-Type': 'application/json' });
                res.end(JSON.stringify({
                    success: true,
                    subtotal,
                    discount,
                    total,
                    receiptId: `REC-${Date.now()}`
                }));
            } catch (err) {
                res.writeHead(400, { 'Content-Type': 'application/json' });
                res.end(JSON.stringify({ error: '無效的結帳資料' }));
            }
        });
        return;
    }

    let safeUrlPath = req.url === '/' ? '/index.html' : req.url;
    let targetFilePath = path.resolve(path.join(PUBLIC_DIR, safeUrlPath));

    let relativePathRelation = path.relative(PUBLIC_DIR, targetFilePath);
    if (relativePathRelation.startsWith('..') || path.isAbsolute(relativePathRelation)) {
        res.writeHead(403, { 'Content-Type': 'text/plain; charset=utf-8' });
        res.end('403 Forbidden: 拒絕目錄越界存取');
        return;
    }

    fs.stat(targetFilePath, (err, stats) => {
        if (err || !stats.isFile()) {
            res.writeHead(404, { 'Content-Type': 'text/plain; charset=utf-8' });
            res.end('404 Not Found: 找不到該檔案');
            return;
        }

        let extension = path.extname(targetFilePath).toLowerCase();
        let contentType = mimeTypes[extension] || 'application/octet-stream';

        const isBinary = extension === '.jpg' || extension === '.png' || extension === '.ico';

        fs.readFile(targetFilePath, isBinary ? null : 'utf-8', (readErr, fileData) => {
            if (readErr) {
                res.writeHead(500, { 'Content-Type': 'text/plain; charset=utf-8' });
                res.end('500 Internal Server Error');
                return;
            }
            res.writeHead(200, { 'Content-Type': contentType });
            res.end(fileData);
        });
    });
});

server.listen(PORT, () => {
    console.log(`✔ 原生 Node.js 商城伺服器已啟動於：http://localhost:${PORT}`);
});
```

```javascript
            }

            sandboxWorker.terminate();
            URL.revokeObjectURL(workerUrl);
```
